# 0.创建python3.10
    conda create -n ml python=3.10
    conda activate ml 
# 1.清理现有环境
     conda activate ml 
     pip uninstall torch torchvision torchaudio -y  
     pip cache purge
# 2.安装cuda 12.8 Nightly版本。安装时会自动匹配相应的torch、torchvision、torchaudio版本(cu128表示12.8)
    pip3 install --pre torch torchvision torchaudio --index-url https://download.pytorch.org/whl/nightly/cu128
    安装完成后：
    Successfully installed torch-2.8.0.dev20250607+cu128 torchaudio-2.8.0.dev20250607+cu128 torchvision-0.23.0.dev20250607+cu128


# 3. 安装transformers
    pip install transformers  自动匹配最新适合的,不适合会降低。
    pip install  transformers==4.52.4   或手动安装指定版本

    下载模型进行测试检查兼容性：
        from transformers import pipeline
        classifier = pipeline("sentiment-analysis")
        print(classifier("I love using transformers!"))
# 4.安装datasets和tokenizers
   pip install transformers datasets tokenizers
